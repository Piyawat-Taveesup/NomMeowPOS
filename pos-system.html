function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Sales") || SpreadsheetApp.getActiveSpreadsheet() .insertSheet("Sales"); // Add headers if empty if (sheet.getLastRow() === 0) { sheet.appendRow([ "Date","Time","Order ID","Customer", "Items","Qty","Unit Price","Total", "Payment","Store" ]); } var data = JSON.parse(e.postData.contents); var rows = data.items.map(function(item) { return [ data.date, data.time, data.orderId, data.customer, item.name, item.qty, item.price, item.total, data.payment, data.store ]; }); rows.forEach(function(row) { sheet.appendRow(row); }); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); }
